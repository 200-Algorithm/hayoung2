import heapq
n=int(input())
m=int(input())
edge=[[] for _ in range(n+1)]
for i in range(m):
    a,b,c=map(int,input().split())
    edge[a].append((b,c))
a,b=map(int,input().split())

def d(a):
    que=[]
    heapq.heappush(que,(a,0))
    dist=[float('inf')]*(n+1)
    dist[a]=0

    while que:
        now,dis=heapq.heappop(que)
        if dist[now] <dis:
            continue
        for next, cost in edge[now]:
            tmp=cost+dis
            if tmp < dist[next]:
                dist[next]=tmp
                heapq.heappush(que,(next,tmp))
    return dist

tmp =d(a)
