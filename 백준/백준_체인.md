# [백준] 체인

1. gpt

```protobuf
start = 0        # 가장 작은 체인을 가리키는 포인터
end = N - 1      # 가장 큰 체인을 가리키는 포인터
cnt = 0          # 작업 횟수(결과값)

# start와 end가 만나기 전까지 반복
while start != end:
    if chains[start] > 0:        # 현재 가장 작은 체인의 길이가 남아있으면
        chains[start] -= 1       # 가장 작은 체인에서 1개 줄임
        end -= 1                 # 가장 큰 체인 1개 제거(연결 완료)
        cnt += 1                 # 연결 횟수 증가
    else:                        # 현재 체인이 0이 되어 더 이상 사용할 수 없으면
        start += 1               # 다음 체인으로 이동

print(cnt)  # 총 연결 횟수 출력
```
