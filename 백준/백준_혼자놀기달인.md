def solution(cards):
    answer = []
    n=len(cards)
    v=[False]*(n+1)
    for card in cards:
        if not v[card]:
            tmp=[]
            while card not in tmp:
                tmp.append(card)
                card=cards[card-1]
                v[card]=True
            print(tmp)
            answer.append(len(tmp))
    
    if answer[0]==n:
        return 0
    return answer[0]*answer[1]
