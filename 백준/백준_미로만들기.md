# [백준] 미로 만들기

1. 성공 푸하하하하ㅏ하핳

```protobuf
n=int(input())
num=list(input().strip())
d=2
# 0 북 1 동 2 남 3 서
arr=[['#'] * 100 for _ in range(100)]
now=[50,50]
arr[50][50]='.'
dx=[[-1,0],[0,1],[1,0],[0,-1]]
maxcol=50
mincol=50
for i in num:
    if i=='R':
        if d!=3:
            d+=1
        else:
            d=0
    elif i=='L':
        if d!=0:
            d-=1
        else:
            d=3
    else:
        now[0]+=dx[d][0]
        now[1]+=dx[d][1]
        if maxcol<now[1]:
            maxcol=now[1]
        if mincol>now[1]:
            mincol=now[1]
        arr[now[0]][now[1]] ='.'

for i in range(100):
    ch=False
    for j in range(mincol,maxcol+1):
        if arr[i][j]=='.':
            ch=True
            break
    if ch:
        print(''.join(arr[i][mincol:maxcol+1]))
```
