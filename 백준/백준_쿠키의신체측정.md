# [백준] 쿠키의 신체측정

1. 성공

```protobuf
n=int(input())
arr=[]
for i in range(n):
  arr.append(list(input().strip()))
cnt=-1
h=[0,0]
la=0
ra=0
ll=0
rl=0
w=0
for i in range(n):
  for j in range(n):
    if arr[i][j]=='*' and cnt==-1:
      h=[i+1+1,j+1]
      cnt=j
      break
  if cnt!=-1:
    for j in range(n):
      if j<cnt:
        if arr[i+1][j]=='*':
          la+=1
      else:
        if arr[i+1][j]=='*':
          ra+=1
    ra-=1
    cnt1=0
    for j in range(i+2,n):
      if arr[j][cnt]=='*':
        w+=1
      else:
        cnt1=j
        break
    for j in range(cnt1,n):
      if arr[j][cnt-1]=='*':
        ll+=1
      if arr[j][cnt+1]=='*':
        rl+=1
    break

print(h[0],h[1])
print(la,ra,w,ll,rl)
```
