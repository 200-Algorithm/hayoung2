1. 성공

```protobuf
n,k=map(int,input().split())
arr=[]
for i in range(n):
    a,b,c,d=map(int,input().split())
    arr.append([a,b,c,d])

arr=sorted(arr,key=lambda x:(x[1],x[2],x[3]) ,reverse=True)
cnt=1
total=1
for i in range(1,n+1):
    x=arr[i-1]
    if i!=1:
        y=arr[i-2]
        if not (x[1]==y[1] and x[2]==y[2] and x[3]==y[3]):
            cnt=i

    if k==x[0]:
        print(cnt)
```
