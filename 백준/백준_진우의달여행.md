# [백준] 진우의 달 여행 (Small)

1. 실패

```protobuf
n,m=map(int,input().split())
arr=[]

for i in range(n):
    arr.append(list(map(int,input().split())))

from collections import deque
def dfs(arr,start):
    que=deque()
    que.append(start)
    tmp=float('inf')
    v = [[[False] * 3 for _ in range(m)] for j in range(n)]
    while que:
        r,c,d,ck=que.popleft()
        v[r][c][ck]=True
        if r==n-1:
            if tmp>d:
                tmp=d
            continue
        cnt=0
        for idx in ([1,-1],[1,0],[1,1]):
            nr,nc=r+idx[0] , c+idx[1]
            if 0<=nr<n and 0<=nc<m and ck !=cnt and v[nr][nc][cnt]==False :
                v[nr][nc][cnt]=True
                que.append([nr,nc,d+arr[nr][nc],cnt])
            cnt+=1

    return tmp

ans=float('inf')
for i in range(m):
    t=dfs(arr,[0,i,arr[0][i],0])
    if ans>t:
        ans=t
print(ans)
```

1. 흐어어어어ㅠㅠㅠㅠ 

```protobuf
import sys

n, m = map(int, sys.stdin.readline().split())
arr = [list(map(int, sys.stdin.readline().split())) for _ in range(n)]
dp = [[[float('inf')] * 3 for _ in range(m)] for _ in range(n)]

# 첫 번째 행 초기화
for j in range(m):
    for k in range(3):
        dp[0][j][k] = arr[0][j]

# 두 번째 행부터 아래로 내려가며 DP 계산
for i in range(1, n):
    for j in range(m):
        # 왼쪽 대각선(0)으로 이동
        if j > 0:
            # 이전 방향이 '아래' (1)이거나 '오른쪽 대각선' (2)인 경우
            dp[i][j][0] = min(dp[i - 1][j - 1][1], dp[i - 1][j - 1][2]) + arr[i][j]

        # 아래(1)로 이동
        # 이전 방향이 '왼쪽 대각선' (0)이거나 '오른쪽 대각선' (2)인 경우
        dp[i][j][1] = min(dp[i - 1][j][0], dp[i - 1][j][2]) + arr[i][j]

        # 오른쪽 대각선(2)으로 이동
        if j < m - 1:
            # 이전 방향이 '왼쪽 대각선' (0)이거나 '아래' (1)인 경우
            dp[i][j][2] = min(dp[i - 1][j + 1][0], dp[i - 1][j + 1][1]) + arr[i][j]

# 마지막 행의 최소값 찾기
ans = float('inf')
for j in range(m):
    ans = min(ans, min(dp[n - 1][j]))
print(ans)
```
