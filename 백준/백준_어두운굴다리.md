# [백준] 어두운 굴다리

1. 실패

```protobuf
n=int(input())
m=int(input())
arr=list(map(int,input().split()))
start=0
end=n
ans=0
while start<=end:
    mid=(start+end)//2
    v=[False]*n
    for i in arr:
        if i-mid>0:
            if i+mid <n:
                for j in range(i-mid,i+mid):
                    if v[j]!=True:
                        v[j]=True
            else:
                for j in range(i-mid,n):
                    if v[j]!=True:
                        v[j]=True
        else:
            if i+mid<n:
                for j in range(0,i+mid):
                    if v[j]!=True:
                        v[j]=True
            else:
                for j in range(0,n):
                    if v[j]!=True:
                        v[j]=True
    for i in range(n):
        if v[i]==False:
            start=mid+1
            break
    else:
        ans=mid
        end=mid-1
print(ans)

```

1. 성공음하하ㅏ

```protobuf
n=int(input())
m=int(input())
arr=list(map(int,input().split()))
start=0
end=n
ans=0
while start<=end:
    mid=(start+end)//2
    t1,t2=-1,-1
    for i in arr:
        if t1==-1 :
            if i-mid > 0:
                break
            else:
                t1=0
        if t2==-1:
            if i+mid <n:
                t2=i+mid
            else:
                t2=n
        else:
            if (i-mid) <= t2:
                t2=max(t2,i+mid)
            else:
                break

    if t1==0 and t2>=n:
        ans=mid
        end=mid-1
    else:
        start=mid+1

print(ans)

```
