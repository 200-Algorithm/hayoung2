1. gpt 모두 해줘야했음 

```protobuf
import sys

def check(s):
    # 길이가 1인 경우, 더 이상 접을 수 없으므로 규칙 만족 (YES)
    if len(s) <= 1:
        return True

    mid = len(s) // 2
    
    # 가운데를 기준으로 왼쪽과 오른쪽의 짝이 반대인지 확인
    for i in range(mid):
        # s[i]와 s[len(s)-1-i]는 서로 접히는 관계
        if s[i] == s[len(s) - 1 - i]:
            return False  # 같은 문자면 규칙 위반

    # 왼쪽 부분과 오른쪽 부분에 대해 재귀적으로 다시 검사
    left_check = check(s[:mid])
    right_check = check(s[mid+1:])
    
    return left_check and right_check

n = int(sys.stdin.readline())
for _ in range(n):
    s = sys.stdin.readline().strip()
    if check(s):
        print('YES')
    else:
        print('NO')
```
