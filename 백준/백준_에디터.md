```protobuf
            arr.append(s[1])
            now+=1
        else:
            arr.insert(now,s[1])
            now+=1
    else:
        if s[0]=='L':
            if now!=0:
                now-=1
        elif s[0]=='D':
            if now!=len(arr):
                now+=1
        elif s[0]=='B':
            if now!=0:
                now-=1
                arr.pop(now)
print(''.join(map(str,arr)))
```

1. gpt.. 

```protobuf
import sys
from collections import deque

# sys.stdin.readline()을 사용하여 입력을 빠르게 처리
arr = sys.stdin.readline().strip()
n = int(sys.stdin.readline())

# 문자열을 left_stack에 넣고, right_stack은 비워둠
left_stack = deque(arr)
right_stack = deque()

# abc L L l=[a] r=[c,b] a+b+c
# abc L P! D L l=[a,b] r=[c] -> l=[a,b,!] r=[c] ->a,b,!c 임 
 
for _ in range(n):
    command = sys.stdin.readline().strip().split()
    
    if command[0] == 'L':
        if left_stack:  # left_stack이 비어있지 않으면
            right_stack.appendleft(left_stack.pop())
    elif command[0] == 'D':
        if right_stack: # right_stack이 비어있지 않으면
            left_stack.append(right_stack.popleft())
    elif command[0] == 'B':
        if left_stack:  # left_stack이 비어있지 않으면
            left_stack.pop()
    elif command[0] == 'P':
        left_stack.append(command[1])

# 최종 문자열 합치기
print(''.join(left_stack) + ''.join(right_stack))
```
