# [백준]  봄버맨

1. 성공

```protobuf
R,C,N=map(int,input().split())
arr=[]
board=[[-1]*C for _ in range(R)]
answer=1
for i in range(R):
    arr.append(list(input()))

# 폭탄설치함 
for i in range(R):
    for j in range(C):
        if arr[i][j]=='O':
            board[i][j]=2 
N-=1 # 1초동안 아무것도안함

while N>0:
# 폭탄 설치 안된 곳 설치하기 
    for i in range(R):
        for j in range(C):
            if arr[i][j]=='.':
                board[i][j]=4
                arr[i][j]='O'

    boom=[]
    for i in range(R):
        for j in range(C):
            if board[i][j]>0:
                board[i][j]-=1
                # 0초되면 터짐 
                if board[i][j]==0:
                # 한번에 삭제하려고 값 추가함 
                    boom.append([i,j])
                    for idx in ([0,1],[1,0],[-1,0],[0,-1]):
                        nr,nc=i+idx[0],j+idx[1]
                        if 0<=nr<R and 0<=nc<C:
                            boom.append([nr,nc])

		# 폭탄! 하면 -1로 설정 
    for nr,nc in boom:
        arr[nr][nc]='.'
        board[nr][nc]=-1

    N-=1
for i in range(R):
    for j in range(C):
        print(arr[i][j],end='')
    print()

```
