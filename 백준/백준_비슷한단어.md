# [백준] 비슷한 단어

1. gpt 다시 하나씩 써보기 

```protobuf
from collections import Counter

n=int(input())
arr=[]
for i in range(n):
    arr.append(input().strip())
base=arr[0] # 첫번째랑만 비교하는거 
alpha=[0]*26

for c in base:
# 아스키코드 
    alpha[ord(c)-ord('A')]+=1

cnt=0
for w in arr[1:]:
    if abs(len(base)-len(w))>1:
        continue
    tmp=alpha[:]
    for c in w:
        tmp[ord(c)-ord('A')]-=1
    diff = sum(abs(x) for x in tmp)
    if diff ==0 or diff==1:
        cnt+=1
    elif diff==2 and len(base)==len(w):
        cnt+=1
print(cnt)
```
