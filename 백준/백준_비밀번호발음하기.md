n=input()
while n!='end':
    n=list(n)
    ck1=False
    ck2=False
    c3=False
    tmp=''
    for i in n:
        if ck1 ==False and i in ['a','e','i','o','u']:
            ck1=True
        tmp+=i
        if len(tmp)!=1:
            ck=tmp[-1]+tmp[-2]
            if  tmp[-1]==tmp[-2] and ck!='ee' and ck!='oo':
                break
            if tmp=='eee' or tmp=='ooo':
                break
            if len(tmp)>2:
                t1=0
                t2=0
                for j in list(tmp):
                    if j in ['a','e','i','o','u']:
                        t1+=1
                    else:
                        t2+=1
                if t1>2 or t2>2:
                    break
                tmp=tmp[1:]
    else:
        ck2=True
        ck3=True
    if ck1 and ck2 and ck3:
        print('<',end='')
        print(''.join(n),end='')
        print('>',end=' ')
        print('is acceptable.')
    else:
        print('<',end='')
        print(''.join(n),end='')
        print('>',end=' ')
        print('is not acceptable.')
    n=input()
