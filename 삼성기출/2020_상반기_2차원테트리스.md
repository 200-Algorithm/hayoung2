1. 정답 보자 열심히 !!!! 

```protobuf
k = int(input())
arr = [[False] * 10 for _ in range(10)]
ans = 0
for idx in range(k):
    # 블록 정보, x,y
    a, b, c = map(int, input().split())

    if a == 1: # 1x1 블록
        # 1. 각 보드에서의 최종 위치를 '먼저' 계산
        # 파란색 보드 (오른쪽으로 이동)
        blue_final_c = 9
        for i in range(c + 1, 10):
            if arr[b][i]:
                blue_final_c = i - 1
                break
        
        # 초록색 보드 (아래로 이동)
        green_final_r = 9
        for i in range(b + 1, 10):
            if arr[i][c]:
                green_final_r = i - 1
                break
        
        # 2. 계산된 최종 위치에 블록을 '한번에' 배치
        arr[b][blue_final_c] = True
        arr[green_final_r][c] = True

    elif a == 2: # 1x2 블록
        # 1. 각 보드에서의 최종 위치를 '먼저' 계산
        # 파란색 보드 (오른쪽으로 이동)
        blue_final_c = 9
        for i in range(c + 2, 10):
            if arr[b][i]:
                blue_final_c = i - 1
                break
        
        # 초록색 보드 (아래로 이동)
        green_final_r = 9
        for i in range(b + 1, 10):
            if arr[i][c] or arr[i][c + 1]:
                green_final_r = i - 1
                break
        
        # 2. 계산된 최종 위치에 블록을 '한번에' 배치
        arr[b][blue_final_c] = True
        arr[b][blue_final_c - 1] = True
        arr[green_final_r][c] = True
        arr[green_final_r][c + 1] = True

    else: # a == 3, 2x1 블록
        # 1. 각 보드에서의 최종 위치를 '먼저' 계산
        # 파란색 보드 (오른쪽으로 이동)
        blue_final_c = 9
        for i in range(c + 1, 10):
            if arr[b][i] or arr[b + 1][i]:
                blue_final_c = i - 1
                break

        # 초록색 보드 (아래로 이동)
        green_final_r = 9
        for i in range(b + 2, 10):
            if arr[i][c]:
                green_final_r = i - 1
                break
        
        # 2. 계산된 최종 위치에 블록을 '한번에' 배치
        arr[b][blue_final_c] = True
        arr[b + 1][blue_final_c] = True
        arr[green_final_r][c] = True
        arr[green_final_r - 1][c] = True

    # 3. 블록 배치 후, 후처리 로직 (파랑 -> 초록 순서)
    # 파란색 보드: 꽉 찬 열이 있는지 확인하고 지우기
    while True:
        cleared = False
        for j in range(6, 10):
            if all(arr[i][j] for i in range(4)): # 열이 꽉 찼으면
                ans += 1
                cleared = True
                for col in range(j, 5, -1):
                    for row in range(4):
                        arr[row][col] = arr[row][col - 1]
                for row in range(4):
                    arr[row][5] = False
        if not cleared:
            break
            
    # 파란색 보드: 연한 칸(열 4, 5)에 블록이 있는지 확인
    push_count_blue = 0
    for j in range(4, 6):
        if any(arr[i][j] for i in range(4)):
            push_count_blue += 1
    if push_count_blue > 0:
        for j in range(9, 5, -1):
            for i in range(4):
                arr[i][j] = arr[i][j - push_count_blue]
        for j in range(4, 6):
            for i in range(4):
                arr[i][j] = False

    # 초록색 보드: 꽉 찬 행이 있는지 확인하고 지우기
    while True:
        cleared = False
        for i in range(6, 10):
            if all(arr[i][j] for j in range(4)): # 행이 꽉 찼으면
                ans += 1
                cleared = True
                for row in range(i, 5, -1):
                    for col in range(4):
                        arr[row][col] = arr[row - 1][col]
                for col in range(4):
                    arr[5][col] = False
        if not cleared:
            break

    # 초록색 보드: 연한 칸(행 4, 5)에 블록이 있는지 확인
    push_count_green = 0
    for i in range(4, 6):
        if any(arr[i][j] for j in range(4)):
            push_count_green += 1
    if push_count_green > 0:
        for i in range(9, 5, -1):
            for j in range(4):
                arr[i][j] = arr[i - push_count_green][j]
        for i in range(4, 6):
            for j in range(4):
                arr[i][j] = False

tile_count = 0
# 초록색 보드 타일 수
for i in range(6, 10):
    for j in range(4):
        if arr[i][j]:
            tile_count += 1
# 파란색 보드 타일 수
for i in range(4):
    for j in range(6, 10):
        if arr[i][j]:
            tile_count += 1

print(ans)
print(tile_count)
```
