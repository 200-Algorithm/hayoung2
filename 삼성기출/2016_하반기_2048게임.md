# 실패…(코드트리 해설)

```python
NUM_MOVES = 5  # 총 5번 이동할 수 있음
NONE = -1      # 아직 합칠 숫자가 없음을 의미하는 상수

# 입력 받기
n = int(input())  # 보드 크기 입력
grid = [
    list(map(int, input().split()))
    for _ in range(n)
]  # 현재 보드 상태

# 작업에 필요한 보조 배열들
next_grid = [
    [0 for _ in range(n)]
    for _ in range(n)
]  # 블록 이동 후 임시 저장용
temp = [
    [0 for _ in range(n)]
    for _ in range(n)
]  # 시뮬레이션 이전 상태 저장용

move_dirs = [0 for _ in range(NUM_MOVES)]  # 이동 방향을 담을 배열 (0: 아래, 1: 오른쪽, 2: 위, 3: 왼쪽)

ans = 0  # 최종 정답 (가장 큰 블록 숫자)

# 현재 grid에서 가장 큰 블록 값을 반환
def get_max_block_num():
    return max([
        grid[i][j]
        for i in range(n)
        for j in range(n)
    ])

# grid를 시계 방향으로 90도 회전
def rotate():
    # next_grid 초기화
    for i in range(n):
        for j in range(n):
            next_grid[i][j] = 0

    # 회전 로직: grid[n - j - 1][i] → next_grid[i][j]
    for i in range(n):
        for j in range(n):
            next_grid[i][j] = grid[n - j - 1][i]

    # 회전된 결과를 다시 grid로 복사
    for i in range(n):
        for j in range(n):
            grid[i][j] = next_grid[i][j]

# 아래 방향으로 블록을 떨어뜨리기
def drop():
    # next_grid 초기화
    for i in range(n):
        for j in range(n):
            next_grid[i][j] = 0

    # 열 기준으로 작업 (j: 열 인덱스)
    for j in range(n):
        keep_num, next_row = NONE, n - 1  # keep_num은 직전 숫자 기억, next_row는 떨어질 위치

        # 아래에서 위로 검사 (i: 행 인덱스)
        for i in range(n - 1, -1, -1):
            if not grid[i][j]:
                continue  # 0이면 무시

            if keep_num == NONE:
                keep_num = grid[i][j]  # 처음 보는 숫자일 경우 저장만 함
            elif keep_num == grid[i][j]:
                # 같은 숫자면 합쳐서 떨어뜨림
                next_grid[next_row][j] = keep_num * 2
                keep_num = NONE
                next_row -= 1
            else:
                # 다른 숫자면 이전 숫자 떨어뜨리고 현재 숫자 keep
                next_grid[next_row][j] = keep_num
                keep_num = grid[i][j]
                next_row -= 1

        # 마지막 남은 숫자도 떨어뜨림
        if keep_num != NONE:
            next_grid[next_row][j] = keep_num
            next_row -= 1

    # next_grid 결과를 grid에 복사
    for i in range(n):
        for j in range(n):
            grid[i][j] = next_grid[i][j]

# move_dir 방향(0: 아래, 1: 오른쪽, 2: 위, 3: 왼쪽)으로 기울이기
def tilt(move_dir):
    # 0: 아래는 그대로, 그 외는 회전해서 아래처럼 만듦
    for _ in range(move_dir):
        rotate()

    drop()  # 아래 방향으로 블록 정리

    # 다시 원래 방향으로 되돌리기
    for _ in range(4 - move_dir):
        rotate()

# 현재 move_dirs로 정해진 방향 시나리오에 따라 게임 시뮬레이션
def simulate():
    global ans

    # 현재 grid 저장
    for i in range(n):
        for j in range(n):
            temp[i][j] = grid[i][j]

    # 정해진 방향대로 한 번씩 이동
    for move_dir in move_dirs:
        tilt(move_dir)

    # 최대 블록 값 갱신
    ans = max(ans, get_max_block_num())

    # grid 복원
    for i in range(n):
        for j in range(n):
            grid[i][j] = temp[i][j]

# 5번의 이동 방향 조합을 재귀로 전부 탐색 (총 4^5 = 1024가지)
def search_max_num(cnt):
    if cnt == NUM_MOVES:
        simulate()  # 방향이 완성되면 시뮬레이션
        return

    for i in range(4):  # 0~3 방향
        move_dirs[cnt] = i
        search_max_num(cnt + 1)

# 탐색 시작
search_max_num(0)
print(ans)  # 최대 블록 숫자 출력

```
