from collections import deque

def solution(board):
    answer = 0
    start = [0, 0]
    end = [0, 0]
    n = len(board)
    m = len(board[0])
    # 왼,오,아래,위
    dx = [0, 0, -1, 1]
    dy = [-1, 1, 0, 0]
    for i in range(n):
        for j in range(m):
            if board[i][j] == 'R':
                start = [i, j]
            elif board[i][j] == 'G':
                end = [i, j]
    que = deque()
    v = set()
    for i in range(4):
        que.append([start[0], start[1], i, 0])
        v.add((start[0], start[1], i))

    while que:
        r, c, d, res = que.popleft()
        if board[r][c] == 'G':

            return res

        if d < 2:
            cke=False
            for i in range(1, m+1):
                nr = r + dx[d]
                nc = c + dy[d]
                if 0 <= nr < n and 0 <= nc < m and board[nr][nc]!='D':
                    r, c = nr, nc
                else:
                    break

            for i in range(4):
                if (r, c, i) not in v:
                    v.add((r, c, i))
                    que.append([r, c, i, res + 1])

        else:
            for i in range(1, n+1):
                nr = r + dx[d] 
                nc = c + dy[d]
                if 0 <= nr < n and 0 <= nc < m and board[nr][nc] != 'D':
                    r, c = nr, nc
                else:
                    break

            for i in range(4):
                if (r, c, i) not in v:
                    v.add((r, c, i))
                    que.append([r, c, i, res + 1])

    return -1
