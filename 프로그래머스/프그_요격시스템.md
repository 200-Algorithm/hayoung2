def solution(targets):
    answer = 1
    targets=sorted(targets,key=lambda x:(x[0],x[1]),reverse=True)
    start=targets[0][0]
    end=targets[0][1]

    for i in range(1,len(targets)):
        if targets[i][0] <= start <targets[i][1] or targets[i][0]<end<targets[i][1]:
            continue
        else:
            answer+=1
            start=targets[i][0]
            end=targets[i][1]
    return answer
