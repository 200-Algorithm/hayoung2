```protobuf
# ⬇️ 기존 while ~ return answer 전체를 이 블록으로 교체

# 1) 캘 수 있는 총 광물 개수만 고려(곡괭이 * 5)
limit = min(sum(picks) * 5, len(minerals))

# 2) 5개씩 청크로 자르고, '돌 곡괭이로 캤을 때' 피로도(=난이도) 계산
groups = []
for i in range(0, limit, 5):
    chunk = minerals[i:i+5]
    d = chunk.count('diamond')
    ir = chunk.count('iron')
    st = len(chunk) - d - ir
    hardness = 25*d + 5*ir + st  # 돌 곡괭이 피로도(난이도)
    groups.append((hardness, d, ir, st))

# 3) 난이도 높은 청크부터 좋은 곡괭이를 배정(그리디 핵심)
groups.sort(reverse=True)

answer = 0
for _, d, ir, st in groups:
    if picks[0] > 0:           # 다이아 곡괭이
        answer += d + ir + st
        picks[0] -= 1
    elif picks[1] > 0:         # 철 곡괭이
        answer += 5*d + ir + st
        picks[1] -= 1
    elif picks[2] > 0:         # 돌 곡괭이
        answer += 25*d + 5*ir + st
        picks[2] -= 1
    else:
        break  # 더 이상 곡괭이 없음

return answer
```
