from collections import defaultdict

def solution(weights):
    tmp=defaultdict(int)
    answer = 0
    
    for i in weights:
        tmp[i]+=1
    a2=defaultdict(int)
    a3=defaultdict(int)
    a4=defaultdict(int)
    
    for key,value in tmp.items():
        a2[key*2]+=value
        a3[key*3]+=value
        a4[key*4]+=value
        #1:1
        answer +=((value)*(value-1))//2

    for key,value in a2.items():
        if key in a3:
            answer+=value * a3[key]
    
    for key,value in a3.items():
        if key in a4:
            answer+=value * a4[key]
    
    for key,value in a2.items():
        if key in a4:
            answer+=value * a4[key]
            
    return answer
