```protobuf
from collections import deque
def solution(storage, requests):
    answer = 0
    n=len(storage)
    m=len(storage[0])
    arr=[]
    for i in range(n):
        arr.append(list(storage[i]))

        
    for req in requests:
        if len(req)==2:
            for i in range(n):
                for j in range(m):
                    if arr[i][j]==req[0]:
                        arr[i][j]=''
        else:
            delete=[]
            que=deque()
            for i in range(n):
                for j in range(m):
                    if arr[i][j]==req:
                        que.append([i,j])
            while que:
                r,c=que.popleft()
                for idx in [[0,1],[1,0],[-1,0],[0,-1]]:
                    nr,nc=r+idx[0],c+idx[1]
                    if 0<=nr<n and 0<=nc<m:
                        if arr[nr][nc]=='':
                            delete.append([r,c])
                            break
                        continue
                    else:
                        delete.append([r,c])
                        break
            for r,c in delete:
                arr[r][c]=''
                
        
            
    for i in range(n):
        for j in range(m):
            if arr[i][j]!='':
                answer+=1
            
                        
                
    return answer
```

```protobuf
from collections import deque
def solution(storage, requests):
    answer = 0
    n=len(storage)
    m=len(storage[0])
    arr=[]
    for i in range(n):
        arr.append(list(storage[i]))

        
    for req in requests:
        if len(req)==2:
            for i in range(n):
                for j in range(m):
                    if arr[i][j]==req[0]:
                        arr[i][j]=''
        else:

            delete=set()
            que=deque()
            v=[[False]*m for _ in range(n)]
            for j in range(m):
                if arr[0][j]==req:
                    delete.add((0,j))
                if arr[n-1][j]==req:
                    delete.add((n-1,j))
                if arr[0][j]=='' and v[0][j]==False:
                    que.append([0,j])
                    v[0][j]=True
                if arr[n-1][j]=='' and v[n-1][j]==False:
                    que.append([n-1,j])
                    v[n-1][j]=True
            
            for i in range(n):
                if arr[i][0]==req:
                    delete.add((i,0))
                if arr[i][m-1]==req:
                    delete.add((i,m-1))
                if arr[i][0]=='' and v[i][0]!=True:
                    que.append([i,0])
                    v[i][0]=True
                if arr[i][m-1]=='' and v[i][m-1]!=True:
                    que.append([i,m-1])
                    v[i][m-1]=True

            while que:
                r,c=que.popleft()
                
                for idx in [[0,1],[1,0],[0,-1],[-1,0]]:
                    nr,nc=r+idx[0],c+idx[0]
                    if 0<=nr<n and 0<=nc<m and not v[nr][nc]:
                        v[nr][nc]=True
                        if arr[nr][nc]=='':
                            que.append([nr,nc])
                        elif arr[nr][nc]==req:
                            delete.add((nr,nc))
            delete=list(delete)
            for r,c in delete:
                arr[r][c]=''
            
                            
    for i in range(n):
        for j in range(m):
            if arr[i][j]!='':
                answer+=1
                        
    return answer
```
