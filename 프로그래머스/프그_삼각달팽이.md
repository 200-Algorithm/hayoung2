def solution(n):
    dx=[1,0,-1] # 아래, 오른쪽 , 위 대각선
    dy=[0,1,-1]
    answer = []
    d=0
    board=[[0]*i for i in range(1,n+1)] # 삼각형
    start = [0, 0]
    board[0][0] = 1
    d = 0
    for i in range(2,(n*(n+1)) //2+1 ):
        nr,nc=start[0]+dx[d],start[1]+dy[d]
        if nr>=n or nc>=len(board[nr]) or board[nr][nc]!=0:
            d=(d+1)%3
            nr,nc=start[0]+dx[d],start[1]+dy[d]
        board[nr][nc]=i
        start=[nr,nc]
        
    
    for i in board:
        for j in i:
            answer.append(j)
    return answer
