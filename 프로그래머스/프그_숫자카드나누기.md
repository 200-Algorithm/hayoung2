1. 75점

```protobuf
def solution(arrayA, arrayB):
    answer = 0
    minv=min(arrayA)
    for i in range(2,minv+1):
        for num in arrayA:
            if num%i==0:
                continue
            break
        else:
            for n in arrayB:
                if n%i==0:
                    break
            else:
                answer=i
    
    for i in range(2,minv+1):
        for num in arrayB:
            if num%i==0:
                continue
            break
        else:
            for n in arrayA:
                if n%i==0:
                    break
            else:
                answer=max(i,answer)
    
    
    return answer
```

```protobuf
import math
def solution(arrayA, arrayB):
    answer = arrayA[0]
    ans=arrayB[0]
    for i in arrayA:
        answer=math.gcd(answer,i)
    
    for i in arrayB:
        if i%answer==0:
            answer=0
            break
    
    for i in arrayB:
        ans=math.gcd(ans,i)
    
    for i in arrayA:
        if i%ans==0:
            ans=0
            break
    
    return max(answer,ans)
```
