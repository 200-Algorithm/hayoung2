from math import isqrt

def solution(r1, r2):
    total = 0  # 총 점의 개수

    # x축을 왼쪽 끝(-r2)부터 오른쪽 끝(r2)까지 스캔
    for x in range(-r2, r2 + 1):
        
        # 현 x좌표에서 바깥 원의 최대 y값
        y_max = isqrt(r2*r2 - x*x)
        
        # 현 x좌표에서 안쪽 원의 최소 y값 (구멍의 시작점)
        t = r1*r1 - x*x
        y_min = 0
        if t > 0:
            y_min = isqrt(t - 1) + 1
        
        # 현재 세로줄에 구멍이 없는 경우
        if y_min == 0:
            # 위(y_max) + 아래(y_max) + 중앙(1) 점들을 더함
            total += (2 * y_max) + 1
        # 현재 세로줄에 구멍이 있는 경우
        else:
            # 위쪽 날개(y_min~y_max)와 아래쪽 날개의 점들을 더함
            total += 2 * (y_max - y_min + 1)
            
    return total
