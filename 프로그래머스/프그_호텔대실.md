def solution(book_time):
    answer = 1
    time=[]
    for idx in book_time:
        start,end=idx[0],idx[1]
        tmps=int(start[0:2])*60+int(start[3:])
        tmpe=int(end[0:2])*60+int(end[3:])+10
        time.append([tmps,tmpe])
    time=sorted(time,key=lambda x:(x[0],x[1]))
    leftover=[time[0][1]]
    print(time)
    for i in range(1,len(time)):
        for j in range(len(leftover)):
            if time[i][0]>=leftover[j]:
                leftover[j]=time[i][1]
                break
        else:
            leftover.append(time[i][1])
            answer+=1
    
    return answer
