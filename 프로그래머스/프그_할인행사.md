from collections import deque
def solution(want, number, discount):
    answer = 0
    v=set()
    n=len(number)
    
    def bfs(numb,day):
        que=deque()
        que.append([numb,day-9])
        while que:
            num,dday=que.popleft()
            if dday==day+1:
                if sum(num)==0:
                    return 1
                else:
                    return 0
                
            for i in range(n):
                if want[i] == discount[dday] and num[i]>0:
                    num[i]-=1
                    break
            que.append((num[:],dday+1))
        return 0 
        
        
    for i in range(9,len(discount)):
        if bfs(number[:],i)==1:
            answer+=1
                    
    return answer

from collections import defaultdict

def solution(weights):
    tmp=defaultdict(int)
    answer = 0
    
    for i in weights:
        tmp[i]+=1
    a2=defaultdict(int)
    a3=defaultdict(int)
    a4=defaultdict(int)
    
    for key,value in tmp.items():
        a2[key*2]+=value
        a3[key*3]+=value
        a4[key*4]+=value
        #1:1
        answer +=((value)*(value-1))//2

    for key,value in a2.items():
        if key in a3:
            answer+=value * a3[key]
    
    for key,value in a3.items():
        if key in a4:
            answer+=value * a4[key]
    
    for key,value in a2.items():
        if key in a4:
            answer+=value * a4[key]
            
    return answer
