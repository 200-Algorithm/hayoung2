from collections import defaultdict
def solution(m, musicinfos):
    answer = ''
    dic=defaultdict(int)
    for musicin in musicinfos:
        start,end,name,music=musicin.split(',')
        result=[]
        result2=[]
        for mic in range(len(m)):
            if '#'==m[mic]:
                result2=result2[:-1]
                result2.append(m[mic-1]+m[mic])
            else:
                result2.append(m[mic])
        
        for mic in range(len(music)):
            if '#'==music[mic]:
                result=result[:-1]
                result.append(music[mic-1]+music[mic])
            else:
                result.append(music[mic])
        
        
        hour=int(end[:2])-int(start[:2])
        
        total=int(end[3:5]) -int(start[3:5]) + hour*60
        answer=[]
        tmp=total
        index=0
        while tmp>0:
            tmp-=1
            answer.append(result[index])
            index=(index+1)%len(result)
        
        if len(answer)<len(result2):
            continue
        else:
            index=0
            mlen=len(result2)
            for i in range(len(answer)):
                if index==mlen:
                    dic[name]=total
                    break
                
                if answer[i]==result2[index]:
                    index+=1
                else:
                    index=0
                    if result2[index]==answer[i]:
                        index+=1
                if index==mlen:
                    dic[name]=total
                    break
            else:
                if index==mlen:
                    dic[name]=total
        
    dic=sorted(dic.items(), key=lambda x:x[1],reverse=True)

    if dic==[]:
        return '(None)'
    else:
        return dic[0][0]
