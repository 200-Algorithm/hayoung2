from collections import deque
def solution(n, k, enemy):
    answer = 0

    que.append([n,0,k,0])
    visited=[0]*(n+1)
    mav=-1
    while que:
        x,i,v,d=que.popleft()
         
        if i<elen:
            if x>=enemy[i] and visited[x]<d+1 :
                que.append([x-enemy[i],i+1,v,d+1])
                visited[x]=d+1
            if v>0 and visited[x]<d+1:
                que.append([x,i+1,v-1,d+1])
                visited[x]=d+1
               
    
    return max(visited)

        heapq.heappush(max_heap,-value)
        
        if total>n:
