```protobuf
        a1, b1, c1 = plans[i+1]
        tmp1 = int(b1[0:2])*60 + int(b1[3:])

            l.appendleft(a)
```

```protobuf
from collections import deque

def solution(plans):
    answer = []
    l = deque()
    plans = sorted(plans, key=lambda x: x[1])
    n = len(plans)
    end = 0
    time=[]
    for i in plans:
        a,b,c=i
        # 시작 시간 계산 
        result=int(b[0:2])*60 + int(b[3:])
        time.append([a,result,int(c)])
    
    
    while end <n-1:
        a,b,c=time[end]
        a2,b2,c2=time[end+1]
        remaining =b2-b # 시작 시간 차이 얼마나 나는지 c 보다 차이나야 a2가 먹을 수 있음. 
        # c와 같거나 c 가 더 작으면 a 는 시간 안에 가능 ! 
        if c<=remaining:
            answer.append(a)
            remaining -=c # 제거하고 a2 시작! 
            while remaining > 0 and len(l)!=0: # 값이 존재하고 , 0이 아닌경우 
                now,value=l.popleft()
                if value<=remaining:  # 미뤄났던 시간이 지금 이 시간보다 적으면 바로 끝내기
                    answer.append(now)
                    remaining -=value # 값 제거 
                else: 
                    l.appendleft([now,value-remaining])
                    remaining=0
        else: # c가 더 크면 미뤄야함 ! a2가 시작해야함             
	        l.appendleft([a,c-remaining])
        
        end+=1
    answer.append(time[-1][0]) # 마지막값 추가 
    
    for i in l: # 나머지 있으면 하나씩 시작 
        answer.append(i[0])
        
    
    
    return answer
```
